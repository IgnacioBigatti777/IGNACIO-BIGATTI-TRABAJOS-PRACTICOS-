def usar_la_fuerza(mochila, contador=0):
    if len(mochila) == 0:
        print("No se encontró el sable de luz.")
        return -1

    objeto = mochila[0]
    if objeto == "sable de luz":
        print("¡Sable de luz encontrado!")
        return contador

    return usar_la_fuerza(mochila[1:], contador + 1)

# Acá está la mochila
mochila = ["comida", "botiquín", "linterna", "sable de luz", "mapa"]

# Acá llamás a la función
objetos_revisados = usar_la_fuerza(mochila)

# Mostramos cuántos objetos se sacaron hasta encontrarlo
if objetos_revisados != -1:
    print("Se sacaron", objetos_revisados, "objetos hasta encontrar el sable.")
